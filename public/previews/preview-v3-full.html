from datetime import datetime, timedelta from typing import Dict, Any from copilot.services.http import get_meflow_client from copilot.auth import get_current_token class ContractQueryService: """合同查询服务""" def __init__(self, data_item_id: str = "contract"): self.data_item_id = data_item_id async def query_by_payment_date( self, start_date: str, end_date: str, status: str = "active", page_size: int = 20, ) -> Dict[str, Any]: """查询指定日期范围内需要付款的合同""" token = get_current_token() filters = [ {"field": "payment_date", "op": "between", "value": [start_date, end_date]}, {"field": "status", "value": status}, ] async with get_meflow_client(token) as client: resp = await client.post( f"/api/v1/data/items/{self.data_item_id}/search", json={"pageNo": 1, "pageSize": page_size, "filters": filters}, ) return resp.json().get("data", {}) async def get_this_month(self) -> Dict[str, Any]: """获取本月需要付款的合同""" now = datetime.now() start = now.replace(day=1).strftime("%Y-%m-%d") if now.month == 12: nxt = now.replace(year=now.year+1, month=1, day=1) else: nxt = now.replace(month=now.month+1, day=1) end = (nxt - timedelta(days=1)).strftime("%Y-%m-%d") return await self.query_by_payment_date(start, end) async def get_overdue(self) -> Dict[str, Any]: """获取逾期未付款的合同""" now = datetime.now().strftime("%Y-%m-%d") return await self.query_by_payment_date("2020-01-01", now)

这个服务封装了合同付款查询的核心逻辑，支持按日期范围筛选和逾期查询。
